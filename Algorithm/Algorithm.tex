\textbf{Example of Factorial of a number}

\begin{algorithm}
\caption{Find the factorial of a number}
\begin{algorithmic}[1]
    \Procedure{Fact}{$n$}
    \State Start
    \State Read n
    \State $Fact \leftarrow 1$
    \State $i \leftarrow 1$
     \While{$i \leq n$}
    \State $Fact = Fact * i $ 
    \State $i = i+1$
      \EndWhile
      \State Display $Fact$
      \State Stop
    \EndProcedure
\end{algorithmic}
\end{algorithm}
